Component 3

Validation, Deduplication, Fraud Filter

Component 3 receives compacted view candidates, not every heartbeat. It produces exactly one durable outcome per candidate and never updates per-video ranking counters directly.

InputViewCandidateEvent or WatchSegmentEvent with rawEventRef.
DedupeEvent-id, view-credit, and watch-segment dedupe.
PublishingOutbox/idempotent publishing avoids dedupe swallowing failed publishes.
DefaultIf trust is uncertain, quarantine rather than affect hot rankings.

Raw durability and compaction happen before Component 3. Component 3 emits valid, rejected, suspicious, or late outcomes into Component 4 topics.

2. Internal Decision Pipeline

The old single event-id dedupe is replaced with distinct dedupe gates for transport retry, valid-view credit, and watch-time segments.

Decision Nodes
flowchart TD
  in[ViewCandidateEvent]
  schema{Schema valid?}
  clock{Event time usable?}
  session{Session trusted?}
  eventDup{Duplicate event id?}
  viewDup{Duplicate view credit?}
  segmentDup{Duplicate watch segment?}
  watch{Watch time plausible?}
  fraud{Fraud risk acceptable?}
  policy{Video rankable?}
  accept[ValidViewEvent]
  reject[RejectedWatchEvent]
  quarantine[SuspiciousWatchEvent]
  late[LateBackfillEvent]

  in --> schema
  schema -- no --> reject
  schema -- yes --> clock
  clock -- late within backfill horizon --> late
  clock -- future or too old --> reject
  clock -- acceptable --> session
  session -- unavailable or unknown --> quarantine
  session -- trusted --> eventDup
  eventDup -- yes --> reject
  eventDup -- no --> viewDup
  viewDup -- duplicate --> reject
  viewDup -- new credit --> segmentDup
  segmentDup -- duplicate segment --> reject
  segmentDup -- new segment --> watch
  watch -- impossible --> quarantine
  watch -- plausible --> fraud
  fraud -- high risk or unavailable --> quarantine
  fraud -- allow --> policy
  policy -- unavailable or not rankable --> reject
  policy -- rankable --> accept

public enum ReservationStatus { ACQUIRED, DUPLICATE, IN_PROGRESS, UNAVAILABLE }

5. Processor Sketch With Safe Publishing

The processor writes an outcome into a durable outbox. A separate publisher drains the outbox into Component 4 with idempotent producer keys. Dedupe reservations are leases, not permanent blind seen-bits.

package com.systemdesign.youtube.topk.validation;

import java.time.Duration;
import java.time.Instant;

public final class Component3Validator {
  private final DedupeStore dedupeStore;
  private final SessionTrustStore sessionTrustStore;
  private final FraudGate fraudGate;
  private final PolicyStore policyStore;
  private final ValidationOutbox outbox;
  private final ValidationConfig config;

  public Component3Validator(
      DedupeStore dedupeStore,
      SessionTrustStore sessionTrustStore,
      FraudGate fraudGate,
      PolicyStore policyStore,
      ValidationOutbox outbox,
      ValidationConfig config
  ) {
    this.dedupeStore = dedupeStore;
    this.sessionTrustStore = sessionTrustStore;
    this.fraudGate = fraudGate;
    this.policyStore = policyStore;
    this.outbox = outbox;
    this.config = config;
  }

  public ValidationOutcome validateAndEnqueue(ViewCandidateEvent event, Instant now) {
    ValidationOutcome outcome = validate(event, now);
    String idempotencyKey = event == null ? "null-event" : "outcome:" + event.eventId();
    outbox.putIfAbsent(idempotencyKey, outcome);
    return outcome;
  }

  public ValidationOutcome validate(ViewCandidateEvent event, Instant now) {
    ValidationOutcome schema = validateSchema(event);
    if (schema != null) return schema;

    ValidationOutcome time = validateEventTime(event, now);
    if (time != null) return time;

    CheckResult trust = sessionTrustStore.checkTrusted(event.sessionIdHash(), event.userOrDeviceHash());
    if (trust.status() == CheckStatus.UNAVAILABLE || trust.status() == CheckStatus.UNKNOWN) {
      return suspicious(event, ReasonCode.SESSION_UNAVAILABLE);
    }
    if (trust.status() == CheckStatus.DENY) return suspicious(event, ReasonCode.UNTRUSTED_SESSION);

    Reservation eventId = dedupeStore.reserve("event:" + event.eventId(), config.eventIdTtl());
    if (eventId.status() == ReservationStatus.UNAVAILABLE) return suspicious(event, eventId.reason());
    if (eventId.status() == ReservationStatus.DUPLICATE) return rejected(event, ReasonCode.DUPLICATE_EVENT_ID);

    String viewCreditKey = viewCreditKey(event);
    Reservation viewCredit = dedupeStore.reserve(viewCreditKey, config.viewCreditTtl());
    if (viewCredit.status() == ReservationStatus.UNAVAILABLE) return suspicious(event, viewCredit.reason());
    if (viewCredit.status() == ReservationStatus.DUPLICATE) return rejected(event, ReasonCode.DUPLICATE_VIEW_CREDIT);

    String segmentKey = watchSegmentKey(event);
    Reservation segment = dedupeStore.reserve(segmentKey, config.watchSegmentTtl());
    if (segment.status() == ReservationStatus.UNAVAILABLE) return suspicious(event, segment.reason());
    if (segment.status() == ReservationStatus.DUPLICATE) return rejected(event, ReasonCode.DUPLICATE_WATCH_SEGMENT);

    long boundedWatchMs = boundedWatchMs(event);
    if (boundedWatchMs < config.minValidWatchMs()) return rejected(event, ReasonCode.IMPOSSIBLE_WATCH_TIME);

    CheckResult fraud = fraudGate.check(event);
    if (fraud.status() == CheckStatus.UNAVAILABLE || fraud.status() == CheckStatus.UNKNOWN) {
      return suspicious(event, ReasonCode.FRAUD_UNAVAILABLE);
    }
    if (fraud.status() == CheckStatus.DENY) return suspicious(event, ReasonCode.FRAUD_RISK);

    CheckResult policy = policyStore.checkRankable(event.videoId());
    if (policy.status() == CheckStatus.UNAVAILABLE || policy.status() == CheckStatus.UNKNOWN) {
      return rejected(event, ReasonCode.POLICY_UNAVAILABLE);
    }
    if (policy.status() == CheckStatus.DENY) return rejected(event, ReasonCode.VIDEO_NOT_RANKABLE);

    ValidViewEvent valid = new ValidViewEvent(
        event.eventId(),
        viewCreditKey,
        segmentKey,
        event.traceId(),
        event.rawEventRef(),
        event.videoId(),
        event.eventTime(),
        event.ingestTime(),
        event.country(),
        event.device(),
        event.sourceRegion(),
        1L,
        boundedWatchMs,
        config.validationVersion(),
        config.ruleConfigVersion(),
        fraudGate.modelVersion()
    );

    return new ValidationOutcome(OutcomeType.VALID, ReasonCode.ACCEPTED, valid, event);
  }

  private ValidationOutcome validateSchema(ViewCandidateEvent event) {
    if (event == null) return new ValidationOutcome(OutcomeType.REJECTED, ReasonCode.MISSING_FIELD, null, null);
    if (blank(event.eventId()) || blank(event.videoId()) || blank(event.sessionIdHash())) return rejected(event, ReasonCode.MISSING_FIELD);
    if (blank(event.userOrDeviceHash()) || event.eventTime() == null || event.ingestTime() == null) return rejected(event, ReasonCode.MISSING_FIELD);
    if (!config.allowedDevices().contains(event.device())) return rejected(event, ReasonCode.INVALID_DEVICE);
    if (event.segmentStartMs() < 0 || event.segmentEndMs() < event.segmentStartMs()) return rejected(event, ReasonCode.IMPOSSIBLE_WATCH_TIME);
    return null;
  }

  private ValidationOutcome validateEventTime(ViewCandidateEvent event, Instant now) {
    if (event.ingestTime().isAfter(now.plus(config.maxFutureSkew()))) return rejected(event, ReasonCode.EVENT_TIME_FUTURE);
    if (event.eventTime().isAfter(event.ingestTime().plus(config.maxFutureSkew()))) return rejected(event, ReasonCode.EVENT_TIME_FUTURE);

    Duration lateness = Duration.between(event.eventTime(), event.ingestTime());
    if (lateness.compareTo(config.backfillAllowedLateness()) > 0) return rejected(event, ReasonCode.EVENT_TOO_OLD);
    if (lateness.compareTo(config.hotPathAllowedLateness()) > 0) return late(event, ReasonCode.LATE_FOR_HOT_RANKING);
    return null;
  }

  private long boundedWatchMs(ViewCandidateEvent event) {
    long segmentMs = event.segmentEndMs() - event.segmentStartMs();
    long maxBySegment = segmentMs + config.maxWatchMsSkew();
    long maxByVideo = Math.max(0L, event.videoDurationMs() - event.segmentStartMs());
    return Math.min(event.reportedWatchMs(), Math.min(maxBySegment, maxByVideo));
  }

  private String viewCreditKey(ViewCandidateEvent event) {
    long startBucketSeconds = event.playbackStartTime().getEpochSecond() / 300;
    return "view:" + event.userOrDeviceHash() + ':' + event.videoId() + ':' + event.sessionIdHash() + ':' + startBucketSeconds;
  }

  private String watchSegmentKey(ViewCandidateEvent event) {
    return "segment:" + event.sessionIdHash() + ':' + event.videoId() + ':' + event.segmentStartMs() + ':' + event.segmentEndMs();
  }

  private static ValidationOutcome rejected(ViewCandidateEvent event, ReasonCode reason) {
    return new ValidationOutcome(OutcomeType.REJECTED, reason, null, event);
  }

  private static ValidationOutcome suspicious(ViewCandidateEvent event, ReasonCode reason) {
    return new ValidationOutcome(OutcomeType.SUSPICIOUS, reason, null, event);
  }

  private static ValidationOutcome late(ViewCandidateEvent event, ReasonCode reason) {
    return new ValidationOutcome(OutcomeType.LATE_BACKFILL, reason, null, event);
  }

  private static boolean blank(String value) {
    return value == null || value.isBlank();
  }
}
